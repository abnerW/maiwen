Docker-CE 18.06 安装手册

简介

容器化基础环境。

安装准备

 $ curl https://releases.rancher.com/install-docker/18.06.sh | sh

修改配置文件

 文件名：/usr/lib/systemd/system/docker.service
 [Unit]
 Description=Docker Application Container Engine
 Documentation=https://docs.docker.com
 After=network-online.target firewalld.service
 Wants=network-online.target

 [Service]
 Type=notify
 #ExecStart=/usr/bin/dockerd --tls=true --tlsverify --tlscacert=/data/cert/ca.crt --tlscert=/data/cert/server.crt --tlskey=/data/cert/server.key -H tcp://172.26.130.100:2376 --storage-driver=overlay2 --storage-opt overlay2.override_kernel_check=true --exec-opt native.cgroupdriver=cgroupfs
 ExecStart=/usr/bin/dockerd -g /app/docker/daemon  --insecure-registry 172.16.0.4:5000 -H unix:///var/run/docker.sock -H tcp://172.16.0.4:2375 --storage-driver=overlay2 --storage-opt overlay2.override_kernel_check=true --exec-opt native.cgroupdriver=cgroupfs
 LimitNOFILE=infinity
 ExecReload=/bin/kill -s HUP $MAINPID
 LimitNPROC=infinity
 LimitCORE=infinity
 #TasksMax=infinity
 TimeoutStartSec=0
 Delegate=yes
 KillMode=process
 Restart=on-failure
 StartLimitBurst=3
 StartLimitInterval=60s

 [Install]
 WantedBy=multi-user.target

重启服务

 $ systemctl daemon-reload
 $ systemctl restart docker
